Create database TOBEBETTER;

CREATE TABLE `TOBEBETTER`.`USER` (

    `ID` VARCHAR(120) NOT NULL,
    `CREATE_DATE` BIGINT NOT NULL,
    `UPDATE_DATE` BIGINT NOT NULL,
    `STATUS` INT NOT NULL,

    `USER_NAME` VARCHAR(45) ,
    `PASSWORD` VARCHAR(32) NOT NULL,
 
  `EMAIL` VARCHAR(45) NOT NULL,
  `OTHER_SYS_ID` VARCHAR(45) NOT NULL,
  `PHONE_NUMBER` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `USER_UNIQUE` (`ID` ASC, `EMAIL` ASC, `UPDATE_DATE` ASC, `CREATE_DATE` ASC,  `USER_NAME` ASC, `PHONE_NUMBER` ASC, `OTHER_SYS_ID` ASC, `STATUS` ASC));


  CREATE TABLE `tobebetter`.`Propose` (
    `ID` VARCHAR(120) NOT NULL,
    `CREATE_DATE` BIGINT NOT NULL,
    `UPDATE_DATE` BIGINT NOT NULL,
    `STATUS` INT NOT NULL,
    `CONNECT` VARCHAR(45) ,
    `MESSAGE` VARCHAR(400) NOT NULL,
    `USER_ID` VARCHAR(120) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE INDEX `Propose_UNIQUE` (`ID` ASC, `UPDATE_DATE` ASC, `CREATE_DATE` ASC, `STATUS` ASC));



  CREATE TABLE `tobebetter`.`UserTask` (
    `ID` VARCHAR(120) NOT NULL,
    `CREATE_DATE` BIGINT NOT NULL,
    `UPDATE_DATE` BIGINT NOT NULL,
    `STATUS` INT NOT NULL,
    `TITLE` VARCHAR(45) ,
    `SECONDS` INT NOT NULL,
    `START_DATE` BIGINT NOT NULL,
    `USER_ID` VARCHAR(120) NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE INDEX `UserTask_UNIQUE` (`ID` ASC, `UPDATE_DATE` ASC, `CREATE_DATE` ASC, `STATUS` ASC, `USER_ID` ASC));
